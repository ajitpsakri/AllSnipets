yarn add tailwindcss -D      //this will install tailwind for us

npx tailwind init tailwind.js --full    //inicializing tailwind and adding tailwind.js

yarn add postcss-cli autoprefixer -D

touch postcss.config.js

add this code to your postcss.config.js ==>
const tailwindcss = require('tailwind');
module.exports{
	plugins:[tailwindcss('./tailwind.js'),require('autoprefixer')]
}
